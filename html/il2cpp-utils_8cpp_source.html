39  static auto logger = getLogger().WithContext("ParameterMatch");
40  il2cpp_functions::Init();
41  if (method->parameters_count != argTypes.size()) {
42  return false;
43  }
44  auto genCount = (method->is_generic && !method->is_inflated) ? method->genericContainer->type_argc : 0;
45  if ((size_t)genCount != genTypes.size()) {
46  logger.warning("Potential method match had wrong number of generics %i (expected %i)",
47  genCount, genTypes.size());
48  return false;
49  }
50  // TODO: supply boolStrictMatch and use type_equals instead of IsConvertible if supplied?
51  for (decltype(method->parameters_count) i = 0; i < method->parameters_count; i++) {
52  auto* paramType = method->parameters[i].parameter_type;
53  if (paramType->type == IL2CPP_TYPE_MVAR) {
54  auto genIdx = paramType->data.genericParameterIndex - method->genericContainer->genericParameterStart;
55  if (genIdx < 0) {

57  } else if (genIdx >= genCount) {
58  logger.warning("ParameterMatch was not supplied enough genTypes to determine type of parameter %i "
59  "(had %i, needed %i)!", i, genCount, genIdx);
60  } else {
61  auto* klass = genTypes.at(genIdx);
62  paramType = (paramType->byref) ? &klass->this_arg : &klass->byval_arg;
63  }
64  }
65  // TODO: just because two parameter lists match doesn't necessarily mean this is the best match...
66  if (!(IsConvertible(paramType, argTypes.at(i)))) {
