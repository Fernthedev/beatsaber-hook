195 uintptr_t findPattern(uintptr_t dwAddress, const char* pattern, uintptr_t dwSearchRangeLen) {
196  #define in_range(x, a, b) (x >= a && x <= b)
197  #define get_bits(x) (in_range((x & (~0x20)), 'A', 'F') ? ((x & (~0x20)) - 'A' + 0xA): (in_range(x, '0', '9') ? x - '0': 0))
198  #define get_byte(x) (get_bits(x[0]) << 4 | get_bits(x[1]))
199 
200  // To avoid a lot of bad match candidates, pre-process wildcards at the front of the pattern
201  uintptr_t skippedStartBytes = 0;
202  while(pattern[0] == '\?') {
203  // see comments below for insight on these numbers
204  pattern += (pattern[1] == '\?') ? 3 : 2;
205  skippedStartBytes++;
206  }
207  uintptr_t match = 0; // current match candidate
208  uintptr_t len = strlen(CRASH_UNLESS(pattern));
209  if (dwSearchRangeLen < len) {
210  return 0;
211  }
212  const char* pat = pattern; // current spot in the pattern
213 
214  for (uintptr_t pCur = dwAddress + skippedStartBytes; pCur < dwAddress + dwSearchRangeLen; pCur++) {
215  // If pat[0] is null char, we are done, or if pat >= pattern + len
216  if (pat >= pattern + len || !pat[0] || !pat[1]) {
217  return match;
218  }
219  // For each byte, if the pattern starts with a ? or the current byte matches:
220  if (pat[0] == '\?' || *(char *)pCur == get_byte(pat)) {
221  // If we do not have a match, begin it
222  if (!match) {
223  match = pCur - skippedStartBytes;
224  }
225  // If our next character is at the end of our pattern, we have a match
226  if (pat + 1 >= pattern + len) {
227  return match;
228  }
229  if (pat[0] != '\?' || pat[1] == '\?') {
230  pat += 3; // advance past "xy " or "?? "
231  } else {
232  pat += 2; // advance past "? "
233  }
234  }
235  else {
236  // reset search position to beginning of the failed match; for loop will begin new search at match + 1
237  if (match) pCur = match + skippedStartBytes;
238  pat = pattern;
239  match = 0;
240  }
241  }
242  return 0;
243 }

264 uintptr_t findUniquePatternInLibil2cpp(bool& multiple, const char* pattern, const char* label) {
265  // Essentially call findUniquePattern for each segment listed in /proc/self/maps
266  std::ifstream procMap("/proc/self/maps");
267  std::string line;
268  uintptr_t match = 0;
269  while (std::getline(procMap, line)) {
270  if (line.find("libil2cpp.so") == std::string::npos) {
271  continue;
272  }
273  auto idx = line.find_first_of('-');
274  if (idx == std::string::npos) {
275  CRASH_UNLESS(false);
276  }
277  auto startAddr = std::stoul(line.substr(0, idx), nullptr, 16);
278  auto spaceIdx = line.find_first_of(' ');
279  if (spaceIdx == std::string::npos) {
280  CRASH_UNLESS(false);
281  }
282  auto endAddr = std::stoul(line.substr(idx + 1, spaceIdx - idx - 1), nullptr, 16);
283  // Permissions are 4 characters

285  if (perms.find('r') != std::string::npos) {
