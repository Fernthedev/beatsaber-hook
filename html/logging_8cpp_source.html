64 void LoggerBuffer::flush() {
65  if (length() == 0) {
66  // If we have nothing to write, exit early.
67  return;
68  }
69  // We can open the file without locking path, because it is only created on initialization.
70  std::ofstream file(path, std::ios::app);
71  if (!file.is_open()) {
72  __android_log_print(Logging::CRITICAL, Logger::get().tag.c_str(), "Could not open file: %s when flushing buffer!", path.c_str());
73  }
74  // Then, iterate over all messages and write each of them to the file.
75  // We already must hold the lock for this call.
76  // Assuming we always append to the END of the list, we could theoretically get away without locking on this call (except for length 1)
77  messageLock.lock();
78  for (; !messages.empty(); messages.pop_front()) {
79  file << messages.front().c_str() << '\n';
80  }
81  messageLock.unlock();
82  file.close();
83 }

102 // Now, we COULD be a lot more reasonable and spawn a thread for each buffer logger
103 // However, I think having one should be fine.
104 // Flushing while holding the bufferMutex means that new loggers take awhile to create (everything else must be flushed)
105 // Flushing itself is pretty quick, new messages aren't allowed to be added while the writing is happening (I'm not sure HOW quick it is)
106 class Consumer {
107  public:
108  void operator()() {
109  // Goal here is that we want to iterate over all of the buffers
110  // For each one, we flush our log to the file specified by the path of that buffer.
111  while (true) {
112  // Lock our bufferMutex
113  Logger::bufferMutex.lock();
114  for (auto* buffer : Logger::buffers) {
115  // For each buffer, we want to flush all of the messages.
116  // However, we want to do so in a fashion that isn't terribly unreasonable.
117  buffer->flush();
118  // Ideally, we thread_yield after each buffer flush (may not need to, though)
119  // std::this_thread::yield();
120  }
121  // Also do the get_global() buffer
122  get_global().flush();
123  Logger::bufferMutex.unlock();
124  // Sleep for a bit without the lock to allow other threads to create loggers and add them
125  std::this_thread::sleep_for(std::chrono::microseconds(500));
126  }
127  }
128 };

290 #define LOG_MAX_CHARS 1000
291 void Logger::log(Logging::Level lvl, std::string str) {
292  if (options.silent) {
293  return;
294  }
295  // Chunk string for logcat buffer
296  if (str.length() > LOG_MAX_CHARS) {
297  std::size_t i = 0;
298  while (i < str.length()) {
299  auto sub = str.substr(i, LOG_MAX_CHARS);
300  auto newline = sub.find('\n');
301  if (newline != std::string::npos) {
302  sub = sub.substr(0, newline);
303  i += newline + 1; // Skip actual newline character
304  } else {
305  i += LOG_MAX_CHARS;
306  }
307  __android_log_write(lvl, tag.c_str(), sub.c_str());
308  }
309  } else {
310  __android_log_write(lvl, tag.c_str(), str.c_str());
311  }
312  if (options.toFile) {
313  // If we want to log to file, we want to write to a shared buffer.
314  // This buffer should be consumed by a separate thread (started if we haven't yet started any consumer)
315  // The overhead of this thread should be pretty minimal, all things considered, even if it handles every Logger instance.
316  // The first thing we need to do is create our buffer, and add it to our buffers (lock while doing so)
317  // Then, we need to start our thread if we haven't started it already
318  // Then, we need to add our data to the buffer (lock while doing so)
319  // The thread needs to consume from these buffers (locks while doing so)
320  auto now = std::chrono::system_clock::now();
321  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()) % 1000;
322  auto in_time = std::chrono::system_clock::to_time_t(now);
323  std::tm bt = *std::localtime(&in_time);
324  std::ostringstream oss;
325  oss << std::put_time(&bt, "%m-%d %H:%M:%S.") << std::setfill('0') << std::setw(3) << ms.count();
326  auto msg = oss.str() + " " + get_level(lvl) + " " + tag + ": " + str.c_str();
327  // __android_log_print(Logging::DEBUG, tag.c_str(), "Logging message: %s to file!", msg.c_str());
328  Logger::bufferMutex.lock();
329  buffer.addMessage(msg);
330  get_global().addMessage(msg);
331  Logger::bufferMutex.unlock();
332  startConsumer();
333  }
