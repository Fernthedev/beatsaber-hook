17 std::optional<std::pair<uint64_t, uint64_t>> DecodeBitMasks(unsigned N, unsigned imms, unsigned immr, unsigned regSize, bool logical) {
18  auto len = HighestSetBit((N << 6) | trunc(~imms, 6), 7);
19  if (len < 1) return std::nullopt;
20 
21  unsigned size = 1 << len;
22  assert(regSize >= size);
23  unsigned levels = size - 1; // a real bitmask of the rightmost "size" bits
24  unsigned R = immr & levels;
25  unsigned S = imms & levels;
26  // For logical immediates an all-ones value of S is reserved since it would generate a useless all-ones result
27  if (logical && (S == levels)) return std::nullopt;
28 
29  auto diff = S - R; // "6-bit subtract with borrow"
30  unsigned d = trunc(diff, len);
31 
32  uint64_t welem = ONES(S + 1); // # of sig bits = size
33  uint64_t telem = ONES(d + 1); // # of sig bits = size
34  welem = ROR(welem, size, R);
35  // Replicate the patterns to fill regSize
36  auto wmask = Replicate<uint64_t>(welem, size, regSize);
37  auto tmask = Replicate<uint64_t>(telem, size, regSize);
38  return std::pair{wmask, tmask};
39 }
