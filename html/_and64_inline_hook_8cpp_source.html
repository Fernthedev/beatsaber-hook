109 static bool __fix_branch_imm(instruction inpp, instruction outpp, context *ctxp)
110 {
111  static constexpr uint32_t mbits = 6u;
112  static constexpr uint32_t mask = 0xfc000000u; // 0b11111100000000000000000000000000
113  static constexpr uint32_t rmask = 0x03ffffffu; // 0b00000011111111111111111111111111
114  static constexpr uint32_t op_b = 0x14000000u; // "b" ADDR_PCREL26
115  static constexpr uint32_t op_bl = 0x94000000u; // "bl" ADDR_PCREL26
116 
117  const uint32_t ins = *(*inpp);
118  const uint32_t opc = ins & mask;
119  switch (opc) {
120  case op_b:
121  case op_bl:
122  {
123  intptr_t current_idx = ctxp->get_and_set_current_index(*inpp, *outpp);
124  int64_t absolute_addr = reinterpret_cast<int64_t>(*inpp) + (static_cast<int32_t>(ins << mbits) >> (mbits - 2u)); // sign-extended
125  int64_t new_pc_offset = static_cast<int64_t>(absolute_addr - reinterpret_cast<int64_t>(*outpp)) >> 2; // shifted
126  bool special_fix_type = ctxp->is_in_fixing_range(absolute_addr);
127  // whether the branch should be converted to absolute jump
128  if (!special_fix_type && llabs(new_pc_offset) >= (rmask >> 1)) {
129  bool b_aligned = (reinterpret_cast<uint64_t>(*outpp + 2) & 7u) == 0u;
130  if (opc == op_b) {
131  if (b_aligned != true) {
132  (*outpp)[0] = A64_NOP;
133  ctxp->reset_current_ins(current_idx, ++(*outpp));
134  } //if
135  (*outpp)[0] = 0x58000051u; // LDR X17, #0x8
136  (*outpp)[1] = 0xd61f0220u; // BR X17
137  memcpy(*outpp + 2, &absolute_addr, sizeof(absolute_addr));
138  *outpp += 4;
139  } else {
140  if (b_aligned == true) {
141  (*outpp)[0] = A64_NOP;
142  ctxp->reset_current_ins(current_idx, ++(*outpp));
143  } //if
144  (*outpp)[0] = 0x58000071u; // LDR X17, #12
145  (*outpp)[1] = 0x1000009eu; // ADR X30, #16
146  (*outpp)[2] = 0xd61f0220u; // BR X17
147  memcpy(*outpp + 3, &absolute_addr, sizeof(absolute_addr));
148  *outpp += 5;
149  } //if
150  } else {
151  if (special_fix_type) {
152  intptr_t ref_idx = ctxp->get_ref_ins_index(absolute_addr);
153  if (ref_idx <= current_idx) {
154  new_pc_offset = static_cast<int64_t>(ctxp->dat[ref_idx].ins - reinterpret_cast<int64_t>(*outpp)) >> 2;
155  } else {
156  ctxp->insert_fix_map(ref_idx, *outpp, 0u, rmask);
157  new_pc_offset = 0;
158  } //if
159  } //if
160 
161  (*outpp)[0] = opc | (new_pc_offset & ~mask);
162  ++(*outpp);
163  } //if
164 
165  ++(*inpp);
166  return ctxp->process_fix_map(current_idx), true;
167  }
168  }
169  return false;
170 }

174 static bool __fix_cond_comp_test_branch(instruction inpp, instruction outpp, context *ctxp)
175 {
176  static constexpr uint32_t lsb = 5u;
177  static constexpr uint32_t lmask01 = 0xff00001fu; // 0b11111111000000000000000000011111
178  static constexpr uint32_t mask0 = 0xff000010u; // 0b11111111000000000000000000010000
179  static constexpr uint32_t op_bc = 0x54000000u; // "b.c" ADDR_PCREL19
180  static constexpr uint32_t mask1 = 0x7f000000u; // 0b01111111000000000000000000000000
181  static constexpr uint32_t op_cbz = 0x34000000u; // "cbz" Rt, ADDR_PCREL19
182  static constexpr uint32_t op_cbnz = 0x35000000u; // "cbnz" Rt, ADDR_PCREL19
183  static constexpr uint32_t lmask2 = 0xfff8001fu; // 0b11111111111110000000000000011111
184  static constexpr uint32_t mask2 = 0x7f000000u; // 0b01111111000000000000000000000000
185  static constexpr uint32_t op_tbz = 0x36000000u; // 0b00110110000000000000000000000000 "tbz" Rt, BIT_NUM, ADDR_PCREL14
186  static constexpr uint32_t op_tbnz = 0x37000000u; // 0b00110111000000000000000000000000 "tbnz" Rt, BIT_NUM, ADDR_PCREL14
187 
188  const uint32_t ins = *(*inpp);
189  uint32_t lmask = lmask01;
190  if ((ins & mask0) != op_bc) {
191  uint32_t opc = ins & mask1;
192  if (opc != op_cbz && opc != op_cbnz) {
193  opc = ins & mask2;
194  if (opc != op_tbz && opc != op_tbnz) {
195  return false;
196  } //if
197  lmask = lmask2;
198  } //if
199  } //if
200 
201  intptr_t current_idx = ctxp->get_and_set_current_index(*inpp, *outpp);
202  int64_t absolute_addr = reinterpret_cast<int64_t>(*inpp) + ((ins & ~lmask) >> (lsb - 2u));
203  int64_t new_pc_offset = static_cast<int64_t>(absolute_addr - reinterpret_cast<int64_t>(*outpp)) >> 2; // shifted
204  bool special_fix_type = ctxp->is_in_fixing_range(absolute_addr);
205  if (!special_fix_type && llabs(new_pc_offset) >= (~lmask >> (lsb + 1))) {
206  if ((reinterpret_cast<uint64_t>(*outpp + 4) & 7u) != 0u) {
207  (*outpp)[0] = A64_NOP;
208  ctxp->reset_current_ins(current_idx, ++(*outpp));
209  } //if
210  (*outpp)[0] = (((8u >> 2u) << lsb) & ~lmask) | (ins & lmask); // B.C #0x8
211  (*outpp)[1] = 0x14000005u; // B #0x14
212  (*outpp)[2] = 0x58000051u; // LDR X17, #0x8
213  (*outpp)[3] = 0xd61f0220u; // BR X17
214  memcpy(*outpp + 4, &absolute_addr, sizeof(absolute_addr));
215  *outpp += 6;
216  } else {
217  if (special_fix_type) {
218  intptr_t ref_idx = ctxp->get_ref_ins_index(absolute_addr);
219  if (ref_idx <= current_idx) {
220  new_pc_offset = static_cast<int64_t>(ctxp->dat[ref_idx].ins - reinterpret_cast<int64_t>(*outpp)) >> 2;
221  } else {
222  ctxp->insert_fix_map(ref_idx, *outpp, lsb, ~lmask);
223  new_pc_offset = 0;
224  } //if
225  } //if
226 
227  (*outpp)[0] = (static_cast<uint32_t>(new_pc_offset << lsb) & ~lmask) | (ins & lmask);
228  ++(*outpp);
229  } //if
230 
231  ++(*inpp);
232  return ctxp->process_fix_map(current_idx), true;
233 }

237 static bool __fix_loadlit(instruction inpp, instruction outpp, context *ctxp)
238 {
239  const uint32_t ins = *(*inpp);
240 
241  // memory prefetch("prfm"), just skip it
242  // http://infocenter.arm.com/help/topic/com.arm.doc.100069_0608_00_en/pge1427897420050.html
243  if ((ins & 0xff000000u) == 0xd8000000u) {
244  ctxp->process_fix_map(ctxp->get_and_set_current_index(*inpp, *outpp));
245  ++(*inpp);
246  return true;
247  } //if
248 
249  static constexpr uint32_t msb = 8u;
250  static constexpr uint32_t lsb = 5u;
251  static constexpr uint32_t mask_30 = 0x40000000u; // 0b01000000000000000000000000000000
252  static constexpr uint32_t mask_31 = 0x80000000u; // 0b10000000000000000000000000000000
253  static constexpr uint32_t lmask = 0xff00001fu; // 0b11111111000000000000000000011111
254  static constexpr uint32_t mask_ldr = 0xbf000000u; // 0b10111111000000000000000000000000
255  static constexpr uint32_t op_ldr = 0x18000000u; // 0b00011000000000000000000000000000 "LDR Wt/Xt, label" | ADDR_PCREL19
256  static constexpr uint32_t mask_ldrv = 0x3f000000u; // 0b00111111000000000000000000000000
257  static constexpr uint32_t op_ldrv = 0x1c000000u; // 0b00011100000000000000000000000000 "LDR St/Dt/Qt, label" | ADDR_PCREL19
258  static constexpr uint32_t mask_ldrsw = 0xff000000u; // 0b11111111000000000000000000000000
259  static constexpr uint32_t op_ldrsw = 0x98000000u; // "LDRSW Xt, label" | ADDR_PCREL19 | load register signed word
260  // LDR S0, #0 | 0b00011100000000000000000000000000 | 32-bit
261  // LDR D0, #0 | 0b01011100000000000000000000000000 | 64-bit
262  // LDR Q0, #0 | 0b10011100000000000000000000000000 | 128-bit
263  // INVALID | 0b11011100000000000000000000000000 | may be 256-bit
264 
265  uint32_t mask = mask_ldr;
266  uintptr_t faligned = (ins & mask_30) ? 7u : 3u;
267  if ((ins & mask_ldr) != op_ldr) {
268  mask = mask_ldrv;
269  if (faligned != 7u)
270  faligned = (ins & mask_31) ? 15u : 3u;
271  if ((ins & mask_ldrv) != op_ldrv) {
272  if ((ins & mask_ldrsw) != op_ldrsw) {
273  return false;
274  } //if
275  mask = mask_ldrsw;
276  faligned = 7u;
277  } //if
278  } //if
279 
280  intptr_t current_idx = ctxp->get_and_set_current_index(*inpp, *outpp);
281  int64_t absolute_addr = reinterpret_cast<int64_t>(*inpp) + ((static_cast<int32_t>(ins << msb) >> (msb + lsb - 2u)) & ~3u);
282  int64_t new_pc_offset = static_cast<int64_t>(absolute_addr - reinterpret_cast<int64_t>(*outpp)) >> 2; // shifted
283  bool special_fix_type = ctxp->is_in_fixing_range(absolute_addr);
284  // special_fix_type may encounter issue when there are mixed data and code
285  if (special_fix_type || (llabs(new_pc_offset) + (faligned + 1u - 4u) / 4u) >= (~lmask >> (lsb + 1))) { // inaccurate, but it works
286  while ((reinterpret_cast<uint64_t>(*outpp + 2) & faligned) != 0u) {
287  *(*outpp)++ = A64_NOP;
288  }
289  ctxp->reset_current_ins(current_idx, *outpp);
290 
291  // Note that if memory at absolute_addr is writeable (non-const), we will fail to fetch it.
292  // And what's worse, we may unexpectedly overwrite something if special_fix_type is true...
293  uint32_t ns = static_cast<uint32_t>((faligned + 1) / sizeof(uint32_t));
294  (*outpp)[0] = (((8u >> 2u) << lsb) & ~mask) | (ins & lmask); // LDR #0x8
295  (*outpp)[1] = 0x14000001u + ns; // B #0xc
296  memcpy(*outpp + 2, reinterpret_cast<void *>(absolute_addr), faligned + 1);
297  *outpp += 2 + ns;
298  } else {
299  faligned >>= 2; // new_pc_offset is shifted and 4-byte aligned
300  while ((new_pc_offset & faligned) != 0) {
301  *(*outpp)++ = A64_NOP;
302  new_pc_offset = static_cast<int64_t>(absolute_addr - reinterpret_cast<int64_t>(*outpp)) >> 2;
303  }
304  ctxp->reset_current_ins(current_idx, *outpp);
305 
306  (*outpp)[0] = (static_cast<uint32_t>(new_pc_offset << lsb) & ~mask) | (ins & lmask);
307  ++(*outpp);
308  } //if
309 
310  ++(*inpp);
311  return ctxp->process_fix_map(current_idx), true;
312 }

316 static bool __fix_pcreladdr(instruction inpp, instruction outpp, context *ctxp)
317 {
318  // Load a PC-relative address into a register
319  // http://infocenter.arm.com/help/topic/com.arm.doc.100069_0608_00_en/pge1427897645644.html
320  static constexpr uint32_t msb = 8u;
321  static constexpr uint32_t lsb = 5u;
322  static constexpr uint32_t mask = 0x9f000000u; // 0b10011111000000000000000000000000
323  static constexpr uint32_t rmask = 0x0000001fu; // 0b00000000000000000000000000011111
324  static constexpr uint32_t lmask = 0xff00001fu; // 0b11111111000000000000000000011111
325  static constexpr uint32_t fmask = 0x00ffffffu; // 0b00000000111111111111111111111111
326  static constexpr uint32_t max_val = 0x001fffffu; // 0b00000000000111111111111111111111
327  static constexpr uint32_t op_adr = 0x10000000u; // "adr" Rd, ADDR_PCREL21
328  static constexpr uint32_t op_adrp = 0x90000000u; // "adrp" Rd, ADDR_ADRP
329 
330  const uint32_t ins = *(*inpp);
331  intptr_t current_idx;
332  switch (ins & mask) {
333  case op_adr:
334  {
335  current_idx = ctxp->get_and_set_current_index(*inpp, *outpp);
336  int64_t lsb_bytes = static_cast<uint32_t>(ins << 1u) >> 30u;
337  int64_t absolute_addr = reinterpret_cast<int64_t>(*inpp) + (((static_cast<int32_t>(ins << msb) >> (msb + lsb - 2u)) & ~3u) | lsb_bytes);
338  int64_t new_pc_offset = static_cast<int64_t>(absolute_addr - reinterpret_cast<int64_t>(*outpp));
339  bool special_fix_type = ctxp->is_in_fixing_range(absolute_addr);
340  if (!special_fix_type && llabs(new_pc_offset) >= (max_val >> 1)) {
341  if ((reinterpret_cast<uint64_t>(*outpp + 2) & 7u) != 0u) {
342  (*outpp)[0] = A64_NOP;
343  ctxp->reset_current_ins(current_idx, ++(*outpp));
344  } //if
345 
346  (*outpp)[0] = 0x58000000u | (((8u >> 2u) << lsb) & ~mask) | (ins & rmask); // LDR #0x8
347  (*outpp)[1] = 0x14000003u; // B #0xc
348  memcpy(*outpp + 2, &absolute_addr, sizeof(absolute_addr));
349  *outpp += 4;
350  } else {
351  if (special_fix_type) {
352  intptr_t ref_idx = ctxp->get_ref_ins_index(absolute_addr & ~3ull);
353  if (ref_idx <= current_idx) {
354  new_pc_offset = static_cast<int64_t>(ctxp->dat[ref_idx].ins - reinterpret_cast<int64_t>(*outpp));
355  } else {
356  ctxp->insert_fix_map(ref_idx, *outpp, lsb, fmask);
357  new_pc_offset = 0;
358  } //if
359  } //if
360 
361  // the lsb_bytes will never be changed, so we can use lmask to keep it
362  (*outpp)[0] = (static_cast<uint32_t>(new_pc_offset << (lsb - 2u)) & fmask) | (ins & lmask);
363  ++(*outpp);
364  } //if
365  }
366  break;
367  case op_adrp:
368  {
369  current_idx = ctxp->get_and_set_current_index(*inpp, *outpp);
370  int32_t lsb_bytes = static_cast<uint32_t>(ins << 1u) >> 30u;
371  int64_t absolute_addr = (reinterpret_cast<int64_t>(*inpp) & ~0xfffll) + ((((static_cast<int32_t>(ins << msb) >> (msb + lsb - 2u)) & ~3u) | lsb_bytes) << 12);
372  A64_LOGI("ins = 0x%.8X, pc = %p, abs_addr = %p",
373  ins, *inpp, reinterpret_cast<int64_t *>(absolute_addr));
374  if (ctxp->is_in_fixing_range(absolute_addr)) {
375  intptr_t ref_idx = ctxp->get_ref_ins_index(absolute_addr/* & ~3ull*/);
376  if (ref_idx > current_idx) {
377  // the bottom 12 bits of absolute_addr are masked out,
378  // so ref_idx must be less than or equal to current_idx!
379  A64_LOGE("ref_idx must be less than or equal to current_idx!");
380  } //if
381 
382  // *absolute_addr may be changed due to relocation fixing
383  A64_LOGI("What is the correct way to fix this?");
384  *(*outpp)++ = ins; // 0x90000000u;
385  } else {
386  if ((reinterpret_cast<uint64_t>(*outpp + 2) & 7u) != 0u) {
387  (*outpp)[0] = A64_NOP;
388  ctxp->reset_current_ins(current_idx, ++(*outpp));
389  } //if
390 
391  (*outpp)[0] = 0x58000000u | (((8u >> 2u) << lsb) & ~mask) | (ins & rmask); // LDR #0x8
392  (*outpp)[1] = 0x14000003u; // B #0xc
393  memcpy(*outpp + 2, &absolute_addr, sizeof(absolute_addr)); // potential overflow?
394  *outpp += 4;
395  } //if
396  }
397  break;
398  default:
399  return false;
400  }
401 
402  ctxp->process_fix_map(current_idx);
403  ++(*inpp);
404  return true;
405 }

408 #define __flush_cache(c, n) __builtin___clear_cache(reinterpret_cast<char *>(c), reinterpret_cast<char *>(c) + n)
409 static void __fix_instructions(uint32_t *__restrict inp, int32_t count, uint32_t *__restrict outp)
410 {
411  context ctx;
412  ctx.basep = reinterpret_cast<int64_t>(inp);
413  ctx.endp = reinterpret_cast<int64_t>(inp + count);
414  memset(ctx.dat, 0, sizeof(ctx.dat));
415  static_assert(sizeof(ctx.dat) / sizeof(ctx.dat[0]) == A64_MAX_INSTRUCTIONS,
416  "please use A64_MAX_INSTRUCTIONS!");
417 #ifndef NDEBUG
418  if (count > A64_MAX_INSTRUCTIONS) {
419  A64_LOGE("too many fixing instructions!");
420  } //if
421 #endif // NDEBUG
422 
423  uint32_t *const outp_base = outp;
424 
425  while (--count >= 0) {
426  if (__fix_branch_imm(&inp, &outp, &ctx)) continue;
427  if (__fix_cond_comp_test_branch(&inp, &outp, &ctx)) continue;
428  if (__fix_loadlit(&inp, &outp, &ctx)) continue;
429  if (__fix_pcreladdr(&inp, &outp, &ctx)) continue;
430 
431  // without PC-relative offset
432  ctx.process_fix_map(ctx.get_and_set_current_index(inp, outp));
433  *(outp++) = *(inp++);
434  }
435 
436  static constexpr uint_fast64_t mask = 0x03ffffffu; // 0b00000011111111111111111111111111
437  auto callback = reinterpret_cast<int64_t>(inp);
438  auto pc_offset = static_cast<int64_t>(callback - reinterpret_cast<int64_t>(outp)) >> 2;
439  if (llabs(pc_offset) >= (mask >> 1)) {
440  if ((reinterpret_cast<uint64_t>(outp + 2) & 7u) != 0u) {
441  outp[0] = A64_NOP;
442  ++outp;
443  } //if
444  outp[0] = 0x58000051u; // LDR X17, #0x8
445  outp[1] = 0xd61f0220u; // BR X17
446  *reinterpret_cast<int64_t *>(outp + 2) = callback;
447  outp += 4;
448  } else {
449  outp[0] = 0x14000000u | (pc_offset & mask); // "B" ADDR_PCREL26
450  ++outp;
451  } //if
452 
453  const uintptr_t total = (outp - outp_base) * sizeof(uint32_t);
454  __flush_cache(outp_base, total); // necessary
455 }
