279 // Replicates the given bits (with M denoting the true number of significant bits) until they fill N bits.
280 template<class To, class From>
281 To Replicate(From bits, int M, int N = sizeof(To) * CHAR_BIT) {
282  CRASH_UNLESS(N % M == 0);
283  To replicatedBits = bits;
284  auto repSize = M;
285  while (repSize * 2 <= N) {
286  replicatedBits |= (replicatedBits << repSize);
287  repSize *= 2;
288  }
289  // The overlap should all match since N % M == 0 means that N - M*a % M == 0
290  replicatedBits |= (replicatedBits << (N - repSize));
291  if (N != M) {
292  Logger::get().debug("Replicate %s * %i = %s", std::bitset<sizeof(From)*CHAR_BIT>(bits).to_string().c_str(), N / M,
293  std::bitset<sizeof(To)*CHAR_BIT>(replicatedBits).to_string().c_str());
294  }
295  return replicatedBits;
296 }
